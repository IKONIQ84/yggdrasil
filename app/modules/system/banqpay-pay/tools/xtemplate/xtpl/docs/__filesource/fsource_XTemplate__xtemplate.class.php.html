  272.     function parse ($bname) {
    273. 

  273.  
    274. 

  274.         if (isset($this->preparsed_blocks[$bname])) {
    275. 

  275.  
    276. 

  276.             $copy = $this->preparsed_blocks[$bname];
    277. 

  277.  
    278. 

  278.         } elseif (isset($this->blocks[$bname])) {
    279. 

  279.  
    280. 

  280.             $copy = $this->blocks[$bname];
    281. 

  281.  
    282. 

  282.         } elseif ($this->_ignore_missing_blocks) {
    283. 

  283.             // ------------------------------------------------------
    284. 

  284.             // NW : 17 Oct 2002. Added default of ignore_missing_blocks
    285. 

  285.             //      to allow for generalised processing where some
    286. 

  286.             //      blocks may be removed from the HTML without the
    287. 

  287.             //      processing code needing to be altered.
    288. 

  288.             // ------------------------------------------------------
    289. 

  289.             // JRC: 3/1/2003 added set error to ignore missing functionality
    290. 

  290.             $this->_set_error("parse: blockname [$bname] does not exist");
    291. 

  291.             return;
    292. 

  292.  
    293. 

  293.         } else {
    294. 

  294.  
    295. 

  295.             $this->_set_error("parse: blockname [$bname] does not exist");
    296. 

  296.         }
    297. 

  297.  
    298. 

  298.         /* from there we should have no more {FILE } directives */
    299. 

  299.         if (!isset($copy)) {
    300. 

  300.             die('Block: ' . $bname);
    301. 

  301.         }
    302. 

  302.  
    303. 

  303.         $copy = preg_replace($this->filevar_delim_nl, '', $copy);
    304. 

  304.  
    305. 

  305.         $var_array = array();
    306. 

  306.  
    307. 

  307.         /* find & replace variables+blocks */
    308. 

  308.         preg_match_all("/" . $this->tag_start_delim . "([A-Za-z0-9\._]+? ?#?.*?)" . $this->tag_end_delim. "/", $copy, $var_array);
    309. 

  309.         $var_array = $var_array[1];
    310. 

  310.  
    311. 

  311.         foreach ($var_array as $k => $v) {
    312. 

  312.  
    313. 

  313.             // Are there any comments in the tags {tag#a comment for documenting the template}
    314. 

  314.             $any_comments = explode('#', $v);
    315. 

  315.             $v = rtrim($any_comments[0]);
    316. 

  316.  
    317. 

  317.             if (sizeof($any_comments) > 1) {
    318. 

  318.  
    319. 

  319.                 $comments = $any_comments[1];
    320. 

  320.             } else {
    321. 

  321.  
    322. 

  322.                 $comments = '';
    323. 

  323.             }
    324. 

  324.  
    325. 

  325.             $sub = explode('.', $v);
    326. 

  326.  
    327. 

  327.             if ($sub[0] == '_BLOCK_') {
    328. 

  328.  
    329. 

  329.                 unset($sub[0]);
    330. 

  330.  
    331. 

  331.                 $bname2 = implode('.', $sub);
    332. 

  332.  
    333. 

  333.                 // trinary operator eliminates assign error in E_ALL reporting
    334. 

  334.                 $var = isset($this->parsed_blocks[$bname2]) ? $this->parsed_blocks[$bname2] : null;
    335. 

  335.                 $nul = (!isset($this->_null_block[$bname2])) ? $this->_null_block[''] : $this->_null_block[$bname2];
    336. 

  336.  
    337. 

  337.                 if ($var == '') {
    338. 

  338.  
    339. 

  339.                     if ($nul == '') {
    340. 

  340.                         // -----------------------------------------------------------
    341. 

  341.                         // Removed requriement for blocks to be at the start of string
    342. 

  342.                         // -----------------------------------------------------------
    343. 

  343.                         //                      $copy=preg_replace("/^\s*\{".$v."\}\s*\n*/m","",$copy);
    344. 

  344.                         // Now blocks don't need to be at the beginning of a line,
    345. 

  345.                         //$copy=preg_replace("/\s*" . $this->tag_start_delim . $v . $this->tag_end_delim . "\s*\n*/m","",$copy);
    346. 

  346.                         $copy = preg_replace("/" . $this->tag_start_delim . $v . $this->tag_end_delim . "/m", '', $copy);
    347. 

  347.  
    348. 

  348.                     } else {
    349. 

  349.  
    350. 

  350.                         $copy = preg_replace("/" . $this->tag_start_delim . $v . $this->tag_end_delim . "/", "$nul", $copy);
    351. 

  351.                     }
    352. 

  352.                 } else {
    353. 

  353.  
    354. 

  354.                     $var = trim($var);
    355. 

  355.                     // SF Bug no. 810773 - thanks anonymous
    356. 

  356.                     //$var = str_replace('\\', '\\\\', $var);
    357. 

  357.                     // Ensure dollars in strings are not evaluated reported by SadGeezer 31/3/04
    358. 

  358.                     //$var = str_replace('$', '\\$', $var);
    359. 

  359.                     // Replaced str_replaces with preg_quote
    360. 

  360.                     $var = preg_quote($var);
    361. 

  361.                     $var = str_replace('\\|', '|', $var);
    362. 

  362.                     $copy = preg_replace("|" . $this->tag_start_delim . $v . $this->tag_end_delim . "|", "$var", $copy);
    363. 

  363.                 }
    364. 

  364.             } else {
    365. 

  365.  
    366. 

  366.                 $var = $this->vars;
    367. 

  367.  
    368. 

  368.                 foreach ($sub as $v1) {
    369. 

  369.  
    370. 

  370.                     // NW 4 Oct 2002 - Added isset and is_array check to avoid NOTICE messages
    371. 

  371.                     // JC 17 Oct 2002 - Changed EMPTY to stlen=0
    372. 

  372.                     //                if (empty($var[$v1])) { // this line would think that zeros(0) were empty - which is not true
    373. 

  373.                     if (!isset($var[$v1]) || (!is_array($var[$v1]) && strlen($var[$v1]) == 0)) {
    374. 

  374.  
    375. 

  375.                         // Check for constant, when variable not assigned
    376. 

  376.                         if (defined($v1)) {
    377. 

  377.  
    378. 

  378.                             $var[$v1] = constant($v1);
    379. 

  379.  
    380. 

  380.                         } else {
    381. 

  381.  
    382. 

  382.                             $var[$v1] = null;
    383. 

  383.                         }
    384. 

  384.                     }
    385. 

  385.  
    386. 

  386.                     $var = $var[$v1];
    387. 

  387.                 }
    388. 

  388.  
    389. 

  389.                 $nul = (!isset($this->_null_string[$v])) ? ($this->_null_string[""]) : ($this->_null_string[$v]);
    390. 

  390.                 $var = (!isset($var)) ? $nul : $var;
    391. 

  391.  
    392. 

  392.                 if ($var == '') {
    393. 

  393.                     // -----------------------------------------------------------
    394. 

  394.                     // Removed requriement for blocks to be at the start of string
    395. 

  395.                     // -----------------------------------------------------------
    396. 

  396.                     //                    $copy=preg_replace("|^\s*\{".$v." ?#?".$comments."\}\s*\n|m","",$copy);
    397. 

  397.                     $copy=preg_replace("|\s*" . $this->tag_start_delim . $v . " ?#?" . $comments . $this->tag_end_delim . "\s*\n|m", '', $copy);
    398. 

  398.                 }
    399. 

  399.  
    400. 

  400.                 $var = trim($var);
    401. 

  401.                 // SF Bug no. 810773 - thanks anonymous
    402. 

  402.                 //$var = str_replace('\\', '\\\\', $var);
    403. 

  403.                 // Ensure dollars in strings are not evaluated reported by SadGeezer 31/3/04
    404. 

  404.                 //$var = str_replace('$', '\\$', $var);
    405. 

  405.                 // Replace str_replaces with preg_quote
    406. 

  406.                 $var = preg_quote($var);
    407.

  673.     /**
    674. 

  674.      * generates the array containing to-be-parsed stuff: $blocks["main"],$blocks["main.table"],$blocks["main.table.row"], etc. also builds the reverse parse order.
    675. 

  675.      *
    676. 

  676.      * @access private
    677. 

  677.      * @param string $con content to be processed
    678. 

  678.      * @param string $parentblock name of the parent block in the block hierarchy
    679. 

  679.      */
    680. 

  680.     function _maketree ($con, $parentblock='') {
    681. 

  681.  
    682. 

  682.         $blocks = array();
    683. 

  683.  
    684. 

  684.         $con2 = explode($this->block_start_delim, $con);
    685. 

  685.  
    686. 

  686.         if (!empty($parentblock)) {
    687. 

  687.  
    688. 

  688.             $block_names = explode('.', $parentblock);
    689. 

  689.             $level = sizeof($block_names);
    690. 

  690.  
    691. 

  691.         } else {
    692. 

  692.  
    693. 

  693.             $block_names = array();
    694. 

  694.             $level = 0;
    695. 

  695.         }
    696. 

  696.  
    697. 

  697.         foreach($con2 as $k => $v) {
    698. 

  698.  
    699. 

  699.             // JRC 06/04/2005 Added block comments (on BEGIN or END) <!-- BEGIN: block_name#Comments placed here -->
    700. 

  700.             //$patt = "($this->block_start_word|$this->block_end_word)\s*(\w+)\s*$this->block_end_delim(.*)";
    701. 

  701.             $patt = "($this->block_start_word|$this->block_end_word)\s*(\w+) ?#?.*?\s*$this->block_end_delim(.*)";
    702. 

  702.  
    703. 

  703.             $res = array();
    704. 

  704.  
    705. 

  705.             if (preg_match_all("/$patt/ims", $v, $res, PREG_SET_ORDER)) {
    706. 

  706.                 // $res[0][1] = BEGIN or END
    707. 

  707.                 // $res[0][2] = block name
    708. 

  708.                 // $res[0][3] = kinda content
    709. 

  709.                 $block_word    = $res[0][1];
    710. 

  710.                 $block_name    = $res[0][2];
    711. 

  711.                 $content    = $res[0][3];
    712. 

  712.                 
    713. 

  713.                 if (strtoupper($block_word) == $this->block_start_word) {
    714. 

  714.  
    715. 

  715.                     $parent_name = implode('.', $block_names);
    716. 

  716.  
    717. 

  717.                     // add one level - array("main","table","row")
    718. 

  718.                     $block_names[++$level] = $block_name;
    719. 

  719.  
    720. 

  720.                     // make block name (main.table.row)
    721. 

  721.                     $cur_block_name=implode('.', $block_names);
    722. 

  722.  
    723. 

  723.                     // build block parsing order (reverse)
    724. 

  724.                     $this->block_parse_order[] = $cur_block_name;
    725. 

  725.  
    726. 

  726.                     //add contents. trinary operator eliminates assign error in E_ALL reporting
    727. 

  727.                     $blocks[$cur_block_name] = isset($blocks[$cur_block_name]) ? $blocks[$cur_block_name] . $content : $content;
    728. 

  728.  
    729. 

  729.                     // add {_BLOCK_.blockname} string to parent block
    730. 

  730.                     $blocks[$parent_name] .= str_replace('\\', '', $this->tag_start_delim) . '_BLOCK_.' . $cur_block_name . str_replace('\\', '', $this->tag_end_delim);
    731. 

  731.  
    732. 

  732.                     // store sub block names for autoresetting and recursive parsing
    733. 

  733.                     $this->sub_blocks[$parent_name][] = $cur_block_name;
    734. 

  734.  
    735. 

  735.                     // store sub block names for autoresetting
    736. 

  736.                     $this->sub_blocks[$cur_block_name][] = '';
    737. 

  737.  
    738. 

  738.                 } else if (strtoupper($block_word) == $this->block_end_word) {
    739. 

  739.  
    740. 

  740.                     unset($block_names[$level--]);
    741. 

  741.  
    742. 

  742.                     $parent_name = implode('.', $block_names);
    743. 

  743.  
    744. 

  744.                     // add rest of block to parent block
    745. 

  745.                     $blocks[$parent_name] .= $res[0][3];
    746. 

  746.                 }
    747. 

  747.             } else {
    748. 

  748.  
    749. 

  749.                 // no block delimiters found
    750. 

  750.                 // Saves doing multiple implodes - less overhead
    751. 

  751.                 $tmp = implode('.', $block_names);
    752. 

  752.  
    753. 

  753.                 if ($k) {
    754. 

  754.                     $blocks[$tmp] .= $this->block_start_delim;
    755. 

  755.                 }
    756. 

  756.  
    757. 

  757.                 // trinary operator eliminates assign error in E_ALL reporting
    758. 

  758.                 $blocks[$tmp] = isset($blocks[$tmp]) ? $blocks[$tmp] . $v : $v;
    759. 

  759.             }
    760. 

  760.         }
    761. 

  761.  
    762. 

  762.         return $blocks;
    763. 

  763.     }
